Differential Expression analysis was performed separately for each cell line, with a design matrix constructed based on the sequencing protocol. For each of the seven cell lines, three count matrices were obtained from the relevant samples in the SummarisedExperiment which holds all the selected samples outlined above. For each of the count matrices (gene counts, transcript counts, and transcript counts obtained based on full length read only), the raw reads were filtered to remove low expression, with min_count = 20 and min_total_count = 50 for genes counts, and min_count = 10, min_total_count = 20 for transcript counts. After filtering, voomWithQualityWeights was utilised to adjust for varying sample quality, and a linear model was fitted with lmFit. Next, limma-voom::decideTest was used with p.value=0.05, and lf = 1.0 or 0.5 for genes and transcripts respectively, and topTables were obtained for each of the comparisons in each of the cell lines.

cell_line <- "A549"

single_cell_line_de_test <- function(all_selected_samples, cell_line, assay_name) {

    cell_line_se <- all_selected_samples[, colData(selected_samples)$cellType ==
        cell_line]

    protocol_ <- colData(cell_line_se)$protocol

    design <- model.matrix(~0 + protocol_)

    # Define a single design for all three contrasts
    protocols_contrasts = makeContrasts(dRNAvsdcDNA = protocol_directRNA - protocol_directcDNA,
        dcDNAvsPCRcDNA = protocol_directcDNA - protocol_PCRcDNA, dRNAvsPCRcDNA = protocol_directRNA -
            protocol_PCRcDNA, levels = colnames(design))

    # Run limma voom with different filter parameters for each assay
    de_test_params <- list(gene_counts = list(lfc = 1, min_count = 20, min_total_count = 50),
        counts = list(lfc = 0.5, min_count = 10, min_total_count = 20), fullLengthCounts = list(lfc = 0.5,
            min_count = 10, min_total_count = 20))

    if (assay_name == "gene_counts") {

        # prep count matrix for DE
        genes_se <- transcriptToGeneExpression(cell_line_se)
        current_counts <- SE2DGEList(genes_se)

    } else {

        current_assay_counts <- assay(cell_line_se, assay_name)
        current_assay_counts[current_assay_counts < 0.5] <- 0
        current_counts <- DGEList(round(current_assay_counts), samples = colData(cell_line_se))
    }
    # run DE test on relevant assay data
    params <- de_test_params[[assay_name]]
    cell_line_DElabels <- do.call(de_filter_fit_test, c(list(current_counts, design,
        protocols_contrasts, return_fit = TRUE), params))  #lfc = 0.5, min_count = 10, min_total_count = 20)

    return(list(de_labels = cell_line_DElabels))


}

Differential Expression analysis was performed separately for each cell line, with a design matrix constructed based on the sequencing protocol. For each of the seven cell lines, three count matrices were obtained from the relevant samples in the SummarisedExperiment which holds all the selected samples outlined above. For each of the count matrices  (gene counts, transcript counts, and transcript counts obtained based on full length read only), the raw reads were filtered to remove low expression, with min_count = 20 and min_total_count = 50 for genes counts, and min_count = 10, min_total_count = 20 for transcript counts.  After filtering, voomWithQualityWeights was utilised to adjust for varying sample quality, and a linear model was fitted with lmFit. Next, limma-voom::decideTest was used with p.value=0.05, and lf = 1.0 or 0.5 for genes and transcripts respectively, and topTables were obtained for each of the comparisons in each of the cell lines.

single_cell_line_de_test <- function(all_selected_samples,cell_line,assay_name) {

    cell_line_se <- all_selected_samples[,colData(selected_samples)$cellType ==  cell_line]
  
    protocol_ <- colData(cell_line_se)$protocol
  
    design <- model.matrix(~0 +  protocol_)
  
    # Define a single design for all three contrasts
    protocols_contrasts = makeContrasts( dRNAvsdcDNA = protocol_directRNA -  protocol_directcDNA,
                                       dcDNAvsPCRcDNA = protocol_directcDNA -  protocol_PCRcDNA,
                                       dRNAvsPCRcDNA = protocol_directRNA -  protocol_PCRcDNA,
                                       levels=colnames(design) )

    
    
    # Run limma voom with different filter parameters for each assay
    de_test_params <- list(
      "gene_counts" = list(lfc= 1., min_count = 20, min_total_count = 50),
      "counts" = list(lfc = 0.5, min_count = 10, min_total_count = 20),
      "fullLengthCounts" = list(lfc = 0.5, min_count = 10, min_total_count = 20)
    )
    
    if (assay_name =="gene_counts"){
      
      # prep count matrix for DE
      genes_se <- transcriptToGeneExpression(cell_line_se)
      current_counts <- SE2DGEList(genes_se)
     
      
    }  else {
      
      current_assay_counts <- assay(cell_line_se ,assay_name)
      current_assay_counts [current_assay_counts < 0.5] <- 0.0
      current_counts <- DGEList(round(current_assay_counts),samples = colData(cell_line_se))
    }
       # run DE test on relevant assay data 
      params <- de_test_params[[assay_name]]
      cell_line_DElabels <- do.call(de_filter_fit_test,c(list(current_counts ,design,protocols_contrasts,return_fit = TRUE),params))#lfc = 0.5, min_count = 10, min_total_count = 20)
   
      
 return(list(de_labels = cell_line_DElabels))   
    
    
}
